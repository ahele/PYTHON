#defined function
def summary(df):
    summ = pd.DataFrame(df.dtypes, columns=['data type'])
    summ['#total'] = df.shape[0]
    summ['#missing'] = df.isnull().sum().values 
    summ['%missing'] = df.isnull().sum().values / len(df)* 100
    summ['#unique'] = df.nunique().values
    summ['#duplicates'] = summ['#total'] - summ['#unique']
    desc = pd.DataFrame(df.describe(include='all').transpose())
    summ['min'] = desc['min'].values
    summ['max'] = desc['max'].values
    return summ

def plot_rates(df):
    plot_vars = df.columns

    fig.legend(loc='center left', bbox_to_anchor=(0.9, 0.5))

def estimate_factors(monthly_changes, factors):
    sensitivities = {}
    for maturity in yields.columns:
        y = monthly_changes[maturity]
        X = sm.add_constant(factors)
        model = sm.OLS(y, X).fit()
        sensitivities[maturity] = model.params
        print(f"Regression results for {maturity}:")
        print(model.summary())
        print("\n")
    return sensitivities

def get_rsquared(monthly_changes, factors):
    r_squared = {}
    for maturity in yields.columns:
        y = monthly_changes[maturity]
        X = sm.add_constant(factors)
        model = sm.OLS(y, X).fit()
        r_squared[maturity] = model.rsquared
    return r_squared
